#version 330
out vec4 FragColor;

in vec2 borderCoord;
in float max_bordercoord;

void main()
{
	float dist = length(borderCoord);	

	if( dist < 1){
		FragColor = vec4(0.0, 0.8, 1.0, 1.0);	
	
	}else if( dist > 1 && dist < max_bordercoord) {
		FragColor = vec4(1.0, 1.0, 1.0, 1.0);
	}else{
		discard;
	}
}
